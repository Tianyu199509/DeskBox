```csharp
// WidgetTrayAnimationController.cs
private const int MaxFPS_HighPriority = 60;   // 动画初期前 100ms
private const int MaxFPS_Normal = 30;          // 后续阶段
private const double HighPriorityDurationMs = 100.0;

private void OnRenderingFrame(object sender, object e)
{
    // 策略：前 100ms 用 60fps（保证流畅），之后降为 30fps
    var elapsed = stopwatch.Elapsed.TotalMilliseconds;
    var targetFPS = elapsed < HighPriorityDurationMs ? MaxFPS_HighPriority : MaxFPS_Normal;
    
    // 跳过不必要的渲染帧
    if (ShouldSkipFrame(targetFPS)) return;
    
    // 原有动画逻辑...
}
```

## ⭐ Phase 2: 方案 C - GPU Translation

### 🎯 目标
**从根本上解决卡顿问题** - 将 CPU 驱动的每帧更新改为 GPU 驱动的 Translation 动画

### 📊 预期收益

| 指标 | 优化前 | 仅方案 A | 方案 C 完成后 |
|------|-------|---------|-------------|
| DWM 调用 | 60 次/秒 | 30-42 次/秒 | **1-2 次/秒** ⬇️97% |
| UI 线程延迟 | 43-123ms | ~25ms | **<5ms** ⬇️80% |
| 帧率稳定性 | 35-45fps | 50-55fps | **60fps 稳定** ⬆️ |
| 总体评分 | 2.5/5 | 3.5/5 | **5/5** ⭐ |

### 🔄 技术原理

```
原方案（CPU 驱动）❌          →      新方案（GPU 驱动）✅
────────────────                  ───────────────────────
每一帧调用 SetWindowPos              1. 设置窗口实际位置（一次）
→ DWM 重绘整个窗口                   2. Translation 模拟滑动效果
→ UI 线程阻塞                        → GPU 硬件加速
→ 60 次/秒调用                       → UI 几乎无负担
```

### 🎨 初步思路（待详细设计）

```csharp
// 目标模式
class WindowUpdateBatcher
{
    private List<WindowUpdate> _pendingUpdates = new();
    
    // 收集更新请求
    public void QueueUpdate(WindowInfo window, Rect newPosition)
    {
        _pendingUpdates.Add(new WindowUpdate(window, newPosition));
    }
    
    // 批量提交（原子性）
    public void Flush()
    {
        // 一次性提交所有更新
        // 避免中间状态的视觉闪烁
        ApplyAllUpdates(_pendingUpdates);
        _pendingUpdates.Clear();
    }
}
```

### 为什么选择 A→C→B 的顺序？

**决策因素：**

1. **风险控制** ✅
   - 方案 A 改动最小，可独立回退
   - 先小步验证，建立信心
   - 每阶段都有明确的验收标准

2. **用户反馈优先** ✅
   - 让用户尽早看到改进
   - 避免长时间投入后发现方向偏差
   - 根据反馈动态调整节奏

3. **技术成熟度** ✅
   - 方案 A 已实现，立即可用
   - 方案 C 有项目内成功案例（DetailPageTransitionHelper）
   - 方案 B 最复杂，需要前期学习和设计

4. **收益最大化** ⭐
   - 方案 A：立竿见影（30-50% 提升）
   - 方案 C：根本解决（80%+ 提升）
   - 方案 B：完善细节（消除残余问题）

### 为什么不按 A→B→C 顺序？

**原因分析：**
- 方案 B 需要深入理解窗口更新机制
- 如果先做方案 B，可能发现与方案 C 冲突
- 方案 C 的收益最大，应优先验证
- 降低返工风险
