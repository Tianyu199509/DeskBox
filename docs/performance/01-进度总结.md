#### 核心代码变更

```csharp
// WidgetTrayAnimationController.cs
// ── 新增智能控制字段 ──
private const int MaxFPS_HighPriority = 60;   // 动画初期前 100ms
private const int MaxFPS_Normal = 30;          // 后续阶段
private const double HighPriorityDurationMs = 100.0;

private DateTime _lastRenderTime;              // 上次渲染时间戳
private TimeSpan _elapsedSinceStart;           // 距离动画开始的时长
private int _targetFPS;                        // 当前目标帧率
private double _minFrameIntervalMs;            // 最小帧间隔 (ms)

// ── OnRenderingFrame 方法中的节流逻辑 ──
_private void OnRenderingFrame(object sender, object e)
{
    // 策略：前 100ms 用 60fps，之后降为 30fps
    _elapsedSinceStart = stopwatch.Elapsed;
    _targetFPS = _elapsedSinceStart.TotalMilliseconds < HighPriorityDurationMs 
        ? MaxFPS_HighPriority 
        : MaxFPS_Normal;
    
    _minFrameIntervalMs = 1000.0 / _targetFPS;
    
    // 跳过不必要的渲染帧
    var timeSinceLastRender = _elapsedSinceStart - (_lastRenderTime - stopwatch.StartTime);
    if (timeSinceLastRender.TotalMilliseconds < _minFrameIntervalMs)
    {
        return;  // 跳过此帧
    }
    
    _lastRenderTime = DateTime.Now;
    // ... 原有逻辑保持不变
}
```
